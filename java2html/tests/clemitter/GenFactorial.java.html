3   import jminusminus.CLEmitter;
4   import static jminusminus.CLConstants.*;
5   import java.util.ArrayList;
6   
7   /**
8    * This class programatically generates the class file for the 
9    * following Java application using CLEmitter:
10   *
11   * This application computes the factorial of a number recursively.
12   *
13   * public class Factorial
14   * {
15   *     public int compute( int n )
16   *     {
17   *         if ( n <= 1 ) {
18   *             return n;
19   *         }
20   *         else {
21   *             return ( n * compute( n - 1 ) );
22   *         }
23   *     }
24   * 
25   *     public static void main( String[] args )
26   *     {
27   *         try {
28   *         Factorial f = new Factorial();
29   *         int n = Int.parseInt( args[ 0 ] );
30   *         System.out.println( "Factorial( " + n + " ) = " + 
31   *                                 f.compute( n ) );
32   *     }
33   *     catch ( NumberFormatException nfe ) {
34   *         System.err.println( "Invalid number " + args[ 0 ] );
35   *     }
36   *     }
37   * }
38   */
39  
40  public class GenFactorial {
41  
42      public static void main(String[] args) {
43          CLEmitter e = new CLEmitter(true);
44          ArrayList<String> accessFlags = new ArrayList<String>();
45  
46          // Add Factorial class
47          accessFlags.add("public");
48          e.addClass(accessFlags, "Factorial", "java/lang/Object", null, true);
49  
50          // Add the implicit no-arg constructor Factorial() to
51          // Factorial
52          accessFlags.clear();
53          accessFlags.add("public");
54          e.addMethod(accessFlags, "<init>", "()V", null, true);
55          e.addNoArgInstruction(ALOAD_0);
56          e.addMemberAccessInstruction(INVOKESPECIAL, "java/lang/Object",
57                  "<init>", "()V");
58          e.addNoArgInstruction(RETURN);
59  
60          // Add compute() method to Factorial
61          accessFlags.clear();
62          accessFlags.add("public");
63          e.addMethod(accessFlags, "compute", "(I)I", null, true);
64          e.addNoArgInstruction(ILOAD_1);
65          e.addNoArgInstruction(ICONST_1);
66          e.addBranchInstruction(IF_ICMPGT, "falseLabel");
67          e.addNoArgInstruction(ILOAD_1);
68          e.addNoArgInstruction(IRETURN);
69          e.addLabel("falseLabel");
70          e.addNoArgInstruction(ILOAD_1);
71          e.addNoArgInstruction(ALOAD_0);
72          e.addNoArgInstruction(ILOAD_1);
73          e.addNoArgInstruction(ICONST_1);
74          e.addNoArgInstruction(ISUB);
75          e.addMemberAccessInstruction(INVOKEVIRTUAL, "Factorial", "compute",
76                  "(I)I");
77          e.addNoArgInstruction(IMUL);
78          e.addNoArgInstruction(IRETURN);
79  
80          // Add main() method to Factorial
81          accessFlags.clear();
82          accessFlags.add("public");
83          accessFlags.add("static");
84          e.addMethod(accessFlags, "main", "([Ljava/lang/String;)V", null, true);
85          e.addExceptionHandler("tryStart", "tryEnd", "catch",
86                  "java/lang/NumberFormatException");
87          e.addLabel("tryStart");
88          e.addReferenceInstruction(NEW, "Factorial");
89          e.addNoArgInstruction(DUP);
90          e.addMemberAccessInstruction(INVOKESPECIAL, "Factorial", "<init>",
91                  "()V");
92          e.addNoArgInstruction(ASTORE_1);
93          e.addNoArgInstruction(ALOAD_0);
94          e.addNoArgInstruction(ICONST_0);
95          e.addNoArgInstruction(AALOAD);
96          e.addMemberAccessInstruction(INVOKESTATIC, "java/lang/Integer",
97                  "parseInt", "(Ljava/lang/String;)I");
98          e.addNoArgInstruction(ISTORE_2);
99          e.addMemberAccessInstruction(GETSTATIC, "java/lang/System", "out",
100                 "Ljava/io/PrintStream;");
101         e.addReferenceInstruction(NEW, "java/lang/StringBuffer");
102         e.addNoArgInstruction(DUP);
103         e.addMemberAccessInstruction(INVOKESPECIAL, "java/lang/StringBuffer",
104                 "<init>", "()V");
105         e.addLDCInstruction("Factorial(");
106         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
107                 "append", "(Ljava/lang/String;)Ljava/lang/StringBuffer;");
108         e.addNoArgInstruction(ILOAD_2);
109         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
110                 "append", "(I)Ljava/lang/StringBuffer;");
111         e.addLDCInstruction(") = ");
112         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
113                 "append", "(Ljava/lang/String;)Ljava/lang/StringBuffer;");
114         e.addNoArgInstruction(ALOAD_1);
115         e.addNoArgInstruction(ILOAD_2);
116         e.addMemberAccessInstruction(INVOKEVIRTUAL, "Factorial", "compute",
117                 "(I)I");
118         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
119                 "append", "(I)Ljava/lang/StringBuffer;");
120         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
121                 "toString", "()Ljava/lang/String;");
122         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/io/PrintStream",
123                 "println", "(Ljava/lang/String;)V");
124         e.addLabel("tryEnd");
125         e.addBranchInstruction(GOTO, "done");
126         e.addLabel("catch");
127         e.addNoArgInstruction(ASTORE_1);
128         e.addMemberAccessInstruction(GETSTATIC, "java/lang/System", "err",
129                 "Ljava/io/PrintStream;");
130         e.addReferenceInstruction(NEW, "java/lang/StringBuffer");
131         e.addNoArgInstruction(DUP);
132         e.addMemberAccessInstruction(INVOKESPECIAL, "java/lang/StringBuffer",
133                 "<init>", "()V");
134         e.addLDCInstruction("Invalid number ");
135         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
136                 "append", "(Ljava/lang/String;)Ljava/lang/StringBuffer;");
137         e.addNoArgInstruction(ALOAD_0);
138         e.addNoArgInstruction(ICONST_0);
139         e.addNoArgInstruction(AALOAD);
140         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
141                 "append", "(Ljava/lang/String;)Ljava/lang/StringBuffer;");
142         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
143                 "toString", "()Ljava/lang/String;");
144         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/io/PrintStream",
145                 "println", "(Ljava/lang/String;)V");
146         e.addLabel("done");
147         e.addNoArgInstruction(RETURN);
148 
149         // Write Factorial.class to file system
150         e.write();
151     }
152
