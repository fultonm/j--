3   import jminusminus.CLEmitter;
4   import static jminusminus.CLConstants.*;
5   import java.util.ArrayList;
6   
7   /**
8    * This class programatically generates the class file for 
9    * the following Java application:
10   * 
11   * public class IntFactorial
12   * {
13   *     public int factorial( int n )
14   *     {
15   *         if ( n &lt;= 1 ) {
16   *             return n;
17   *         }
18   *         else {
19   *             return ( n * factorial( n - 1 ) );
20   *         }
21   *     }
22   *     
23   *     public static void main( String[] args )
24   *     {
25   *         IntFactorial f = new IntFactorial();
26   *         int n = Integer.parseInt( args[ 0 ] );
27   *         System.out.println( "Factorial( " + n + 
28   *             " ) = " + f.factorial( n ) );
29   *     }
30   * }
31   */
32  
33  public class GenIntFactorial {
34  
35      public static void main(String[] args) {
36          CLEmitter e = new CLEmitter(true);
37          ArrayList<String> accessFlags = new ArrayList<String>();
38  
39          // Add IntFactorial class
40          accessFlags.add("public");
41          e.addClass(accessFlags, "IntFactorial", "java/lang/Object", null, true);
42  
43          // Add the implicit no-arg constructor IntFactorial()
44          accessFlags.clear();
45          accessFlags.add("public");
46          e.addMethod(accessFlags, "<init>", "()V", null, true);
47          e.addNoArgInstruction(ALOAD_0);
48          e.addMemberAccessInstruction(INVOKESPECIAL, "java/lang/Object",
49                  "<init>", "()V");
50          e.addNoArgInstruction(RETURN);
51  
52          // Add factorial() method to IntFactorial
53          accessFlags.clear();
54          accessFlags.add("public");
55          e.addMethod(accessFlags, "factorial", "(I)I", null, true);
56          e.addNoArgInstruction(ILOAD_1);
57          e.addNoArgInstruction(ICONST_1);
58          e.addBranchInstruction(IF_ICMPGT, "Label1");
59          e.addNoArgInstruction(ILOAD_1);
60          e.addNoArgInstruction(IRETURN);
61          e.addLabel("Label1");
62          e.addNoArgInstruction(ILOAD_1);
63          e.addNoArgInstruction(ALOAD_0);
64          e.addNoArgInstruction(ILOAD_1);
65          e.addNoArgInstruction(ICONST_1);
66          e.addNoArgInstruction(ISUB);
67          e.addMemberAccessInstruction(INVOKEVIRTUAL, "IntFactorial",
68                  "factorial", "(I)I");
69          e.addNoArgInstruction(IMUL);
70          e.addNoArgInstruction(IRETURN);
71  
72          // Add main() method to IntFactorial
73          accessFlags.clear();
74          accessFlags.add("public");
75          accessFlags.add("static");
76          e.addMethod(accessFlags, "main", "([Ljava/lang/String;)V", null, true);
77          e.addReferenceInstruction(NEW, "IntFactorial");
78          e.addNoArgInstruction(DUP);
79          e.addMemberAccessInstruction(INVOKESPECIAL, "IntFactorial", "<init>",
80                  "()V");
81          e.addNoArgInstruction(ASTORE_1);
82          e.addNoArgInstruction(ALOAD_0);
83          e.addNoArgInstruction(ICONST_0);
84          e.addNoArgInstruction(AALOAD);
85          e.addMemberAccessInstruction(INVOKESTATIC, "java/lang/Integer",
86                  "parseInt", "(Ljava/lang/String;)I");
87          e.addNoArgInstruction(ISTORE_2);
88          e.addMemberAccessInstruction(GETSTATIC, "java/lang/System", "out",
89                  "Ljava/io/PrintStream;");
90          e.addReferenceInstruction(NEW, "java/lang/StringBuffer");
91          e.addNoArgInstruction(DUP);
92          e.addMemberAccessInstruction(INVOKESPECIAL, "java/lang/StringBuffer",
93                  "<init>", "()V");
94          e.addLDCInstruction("Factorial(");
95          e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
96                  "append", "(Ljava/lang/String;)Ljava/lang/StringBuffer;");
97          e.addNoArgInstruction(ILOAD_2);
98          e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
99                  "append", "(I)Ljava/lang/StringBuffer;");
100         e.addLDCInstruction(") = ");
101         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
102                 "append", "(Ljava/lang/String;)Ljava/lang/StringBuffer;");
103         e.addNoArgInstruction(ALOAD_1);
104         e.addNoArgInstruction(ILOAD_2);
105         e.addMemberAccessInstruction(INVOKEVIRTUAL, "IntFactorial",
106                 "factorial", "(I)I");
107         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
108                 "append", "(I)Ljava/lang/StringBuffer;");
109         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/lang/StringBuffer",
110                 "toString", "()Ljava/lang/String;");
111         e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/io/PrintStream",
112                 "println", "(Ljava/lang/String;)V");
113         e.addNoArgInstruction(RETURN);
114 
115         // Write IntFactorial.class to file system
116         e.write();
117     }
