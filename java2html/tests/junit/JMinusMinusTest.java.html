3   package tests.junit;
4   
5   import java.io.File;
6   import junit.framework.TestCase;
7   import jminusminus.Main;
8   
9   /**
10   * JUnit test case for running the j-- compiler on the j-- test programs under
11   * tests/pass and tests/fail folders.
12   */
13  
14  public class JMinusMinusTest extends TestCase {
15  
16      /**
17       * Construct a JMinusMinusTest object.
18       */
19  
20      public JMinusMinusTest() {
21          super("JUnit test case for the j-- compiler");
22      }
23  
24      /**
25       * Run the j-- compiler against each pass-test file under the folder
26       * specified by PASS_TESTS_DIR property in the build.xml file. FRONT_END
27       * property determines the frontend (handwritten or JavaCC) to use.
28       */
29  
30      public void testPass() {
31          File passTestsDir = new File(System.getProperty("PASS_TESTS_DIR"));
32          File genClassDir = new File(System.getProperty("GEN_CLASS_DIR"));
33          File[] files = passTestsDir.listFiles();
34          boolean errorHasOccurred = false;
35          for (int i = 0; files != null && i < files.length; i++) {
36              if (files[i].toString().endsWith(".java")) {
37                  String[] args = null;
38                  System.out.printf("Running j-- (with "
39                          + "handwritten frontend) on %s ...\n\n", files[i]
40                          .toString());
41                  args = new String[] { "-d", genClassDir.getAbsolutePath(),
42                          files[i].toString() };
43                  Main.main(args);
44                  System.out.printf("\n\n");
45  
46                  // true even if a single test fails
47                  errorHasOccurred |= Main.errorHasOccurred();
48              }
49          }
50  
51          // We want all tests to pass
52          assertFalse(errorHasOccurred);
53      }
54  
55      /**
56       * Run the j-- compiler against each fail-test file under the folder
57       * specified by FAIL_TESTS_DIR property in the build.xml file. FRONT_END
58       * property determines the frontend (handwritten or JavaCC) to use.
59       */
60  
61      public void testFail() {
62          File failTestsDir = new File(System.getProperty("FAIL_TESTS_DIR"));
63          File genClassDir = new File(System.getProperty("GEN_CLASS_DIR"));
64          File[] files = failTestsDir.listFiles();
65          boolean errorHasOccurred = true;
66          for (int i = 0; files != null && i < files.length; i++) {
67              if (files[i].toString().endsWith(".java")) {
68                  String[] args = null;
69                  System.out.printf("Running j-- (with "
70                          + "handwritten frontend) on %s ...\n\n", files[i]
71                          .toString());
72                  args = new String[] { "-d", genClassDir.getAbsolutePath(),
73                          files[i].toString() };
74                  Main.main(args);
75                  System.out.printf("\n\n");
76  
77                  // true only if all tests fail
78                  errorHasOccurred &= Main.errorHasOccurred();
79              }
80          }
81  
82          // We want all tests to fail
83          assertTrue(errorHasOccurred);
84      }
85  
86      /**
87       * Entry point.
88       * 
89       * @param args
90       *            command-line arguments.
91       */
92  
93      public static void main(String[] args) {
94          junit.textui.TestRunner.run(JMinusMinusTest.class);
95      }
96  
97  }
