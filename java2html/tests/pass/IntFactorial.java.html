5   import java.lang.Integer;
6   import java.lang.System;
7   
8   public class IntFactorial {
9   
10      public int factorial(int n) {
11          if (n <= 1) {
12              return n;
13          } else {
14              return (n * factorial(n - 1));
15          }
16      }
17  
18      public static void main(String[] args) {
19          IntFactorial f = new IntFactorial();
20          int n = Integer.parseInt(args[0]);
21          System.out.println("Factorial( " + n + " ) = " + f.factorial(n));
22      }

24  }
