7   /**
8    * The AST node for an assignment statement. This is an abtract class into which
9    * we factor behavior common to all assignment operations.
10   */
11  
12  abstract class JAssignment extends JBinaryExpression {
13  
14      /**
15       * Construct an AST node for an assignment operation.
16       * 
17       * @param line     line in which the assignment operation occurs in the source
18       *                 file.
19       * @param operator the actual assignment operator.
20       * @param lhs      the lhs operand.
21       * @param rhs      the rhs operand.
22       */
23  
24      public JAssignment(int line, String operator, JExpression lhs, JExpression rhs) {
25          super(line, operator, lhs, rhs);
26      }
27  
28  }
29  
30  /**
31   * The AST node for an assignment (=) expression. The = operator has two
32   * operands: a lhs and a rhs.
33   */
34  
35  class JAssignOp extends JAssignment {
36  
37      /**
38       * Construct the AST node for an assignment (=) expression given the lhs and rhs
39       * operands.
40       * 
41       * @param line line in which the assignment expression occurs in the source
42       *             file.
43       * @param lhs  lhs operand.
44       * @param rhs  rhs operand.
45       */
46  
47      public JAssignOp(int line, JExpression lhs, JExpression rhs) {
48          super(line, "=", lhs, rhs);
49      }
50  
51      /**
52       * Analyze the lhs and rhs, checking that types match, and set the result type.
53       * 
54       * @param context context in which names are resolved.
55       * @return the analyzed (and possibly rewritten) AST subtree.
56       */
57  
58      public JExpression analyze(Context context) {
59          if (!(lhs instanceof JLhs)) {
60              JAST.compilationUnit.reportSemanticError(line(), "Illegal lhs for assignment");
61          } else {
62              lhs = (JExpression) ((JLhs) lhs).analyzeLhs(context);
63          }
64          rhs = (JExpression) rhs.analyze(context);
65          rhs.type().mustMatchExpected(line(), lhs.type());
66          type = rhs.type();
67          if (lhs instanceof JVariable) {
68              IDefn defn = ((JVariable) lhs).iDefn();
69              if (defn != null) {
70                  // Local variable; consider it to be initialized now.
71                  ((LocalVariableDefn) defn).initialize();
72              }
73          }
74          return this;
75      }
76  
77      /**
78       * Code generation for an assignment involves, generating code for loading any
79       * necessary Lvalue onto the stack, for loading the Rvalue, for (unless a
80       * statement) copying the Rvalue to its proper place on the stack, and for doing
81       * the store.
82       * 
83       * @param output the code emitter (basically an abstraction for producing the
84       *               .class file).
85       */
86  
87      public void codegen(CLEmitter output) {
88          ((JLhs) lhs).codegenLoadLhsLvalue(output);
89          rhs.codegen(output);
90          if (!isStatementExpression) {
91              // Generate code to leave the Rvalue atop stack
92              ((JLhs) lhs).codegenDuplicateRvalue(output);
93          }
94          ((JLhs) lhs).codegenStore(output);
95      }
96  
97  }

99  /**
100  * The AST node for a += expression. A += expression has two operands: a lhs and
101  * a rhs.
102  */
103 
104 class JPlusAssignOp extends JAssignment {
105 
106     /**
107      * Construct the AST node for a += expression given its lhs and rhs operands.
108      * 
109      * @param line line in which the assignment expression occurs in the source
110      *             file.
111      * @param lhs  the lhs operand.
112      * @param rhs  the rhs operand.
113      */
114 
115     public JPlusAssignOp(int line, JExpression lhs, JExpression rhs) {
116         super(line, "+=", lhs, rhs);
117     }
118 
119     /**
120      * Analyze the lhs and rhs, rewrite rhs as lhs + rhs (string concatenation) if
121      * lhs is a String, and set the result type.
122      * 
123      * @param context context in which names are resolved.
124      * @return the analyzed (and possibly rewritten) AST subtree.
125      */
126 
127     public JExpression analyze(Context context) {
128         if (!(lhs instanceof JLhs)) {
129             JAST.compilationUnit.reportSemanticError(line(), "Illegal lhs for assignment");
130             return this;
131         } else {
132             lhs = (JExpression) ((JLhs) lhs).analyzeLhs(context);
133         }
134         rhs = (JExpression) rhs.analyze(context);
135         if (lhs.type().equals(Type.INT)) {
136             rhs.type().mustMatchExpected(line(), Type.INT);
137             type = Type.INT;
138         } else if (lhs.type().equals(Type.STRING)) {
139             rhs = (new JStringConcatenationOp(line, lhs, rhs)).analyze(context);
140             type = Type.STRING;
141         } else {
142             JAST.compilationUnit.reportSemanticError(line(),
143                     "Invalid lhs type for +=: " + lhs.type());
144         }
145         return this;
146     }
147 
148     /**
149      * Code generation for += involves, generating code for loading any necessary
150      * l-value onto the stack, for (unless a string concatenation) loading the
151      * r-value, for (unless a statement) copying the r-value to its proper place on
152      * the stack, and for doing the store.
153      * 
154      * @param output the code emitter (basically an abstraction for producing the
155      *               .class file).
156      */
157 
158     public void codegen(CLEmitter output) {
159         ((JLhs) lhs).codegenLoadLhsLvalue(output);
160         if (lhs.type().equals(Type.STRING)) {
161             rhs.codegen(output);
162         } else {
163             ((JLhs) lhs).codegenLoadLhsRvalue(output);
164             rhs.codegen(output);
165             output.addNoArgInstruction(IADD);
166         }
167         if (!isStatementExpression) {
168             // Generate code to leave the r-value atop stack
169             ((JLhs) lhs).codegenDuplicateRvalue(output);
170         }
171         ((JLhs) lhs).codegenStore(output);
172     }
173 }

704      * r-value, for (unless a statement) copying the r-value to its proper place on
705      * the stack, and for doing the store.
706      * 
707      * @param output the code emitter (basically an abstraction for producing the
708      *               .class file).
709      */
710 
711     public void codegen(CLEmitter output) {
712         ((JLhs) lhs).codegenLoadLhsLvalue(output);
713         ((JLhs) lhs).codegenLoadLhsRvalue(output);
714         rhs.codegen(output);
715         output.addNoArgInstruction(IXOR);
716         if (!isStatementExpression) {
717             // Generate code to leave the r-value atop stack
718             ((JLhs) lhs).codegenDuplicateRvalue(output);
719         }
720         ((JLhs) lhs).codegenStore(output);
721     }
722 }

724 /**
725  * The AST node for a |= expression. A |= expression has two operands: a lhs
726  * and a rhs.
727  */

729 class JBWOrAssignOp extends JAssignment {
730 
731     /**
732      * Construct the AST node for a |= expression given its lhs and rhs operands.
733      * 
734      * @param line line in which the assignment expression occurs in the source
735      *             file.
736      * @param lhs  the lhs operand.
737      * @param rhs  the rhs operand.
738      */
739 
740     public JBWOrAssignOp(int line, JExpression lhs, JExpression rhs) {
741         super(line, "|=", lhs, rhs);
742     }
743 
744     /**
745      * Analyze the lhs and rhs, rewrite rhs as lhs | rhs
746      * 
747      * @param context context in which names are resolved.
748      * @return the analyzed (and possibly rewritten) AST subtree.
749      */
750 
751     public JExpression analyze(Context context) {
752         if (!(lhs instanceof JLhs)) {
753             JAST.compilationUnit.reportSemanticError(line(), "Illegal lhs for assignment");
754             return this;
755         } else {
756             lhs = (JExpression) ((JLhs) lhs).analyzeLhs(context);
757         }
758         rhs = (JExpression) rhs.analyze(context);
759         if (lhs.type().equals(Type.INT)) {
760             rhs.type().mustMatchExpected(line(), Type.INT);
761             type = Type.INT;
762         } else {
763             JAST.compilationUnit.reportSemanticError(line(),
764                     "Invalid lhs type for |=: " + lhs.type());
765         }
766         return this;
767     }
768 
769     /**
770      * Code generation for |= involves, generating code for loading any necessary
771      * l-value onto the stack, for (unless a string concatenation) loading the
772      * r-value, for (unless a statement) copying the r-value to its proper place on
773      * the stack, and for doing the store.
774      * 
775      * @param output the code emitter (basically an abstraction for producing the
776      *               .class file).
777      */
778 
779     public void codegen(CLEmitter output) {
780         ((JLhs) lhs).codegenLoadLhsLvalue(output);
781         ((JLhs) lhs).codegenLoadLhsRvalue(output);
782         rhs.codegen(output);
783         output.addNoArgInstruction(IOR);
784         if (!isStatementExpression) {
785             // Generate code to leave the r-value atop stack
786             ((JLhs) lhs).codegenDuplicateRvalue(output);
787         }
788         ((JLhs) lhs).codegenStore(output);
789     }
