3   import spim.SPIM;
4   
5   // A test to check if argument passing is done correctly with calls 
6   // to functions with more than 4 arguments.
7   
8   public class Formals {
9   
10      // Return the sum of the arguments and a local variable.
11  
12      public static int sum(int a, int b, int c, int d, int e, int f) {
13          int g = 7;
14          return a + b + c + d + e + f + g;
15      }
16  
17      // Return the product of the arguments and a local variable.
18  
19      public static int product(int a, int b, int c, int d, int e, int f) {
20          int g = 7;
21          return a * b * c * d * e * f * g;
22      }
23  
24      // Entry point; calls sum() and product() to print the sum
25      // and product of the first 7 integers.
26  
27      public static void main(String[] args) {
28          SPIM.printInt(Formals.sum(1, 2, 3, 4, 5, 6));
29          SPIM.printChar('\n');
30          SPIM.printInt(Formals.product(1, 2, 3, 4, 5, 6));
31          SPIM.printChar('\n');
32      }
33  
34  }
