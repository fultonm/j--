3   package tests.junit;
4   
5   import junit.framework.TestCase;
6   import jminusminus.Main;
7   import java.io.File;
8   
9   /**
10   * JUnit test case for analysis of the AST.
11   */
12  
13  public class AnalysisTest extends TestCase {
14  
15      /**
16       * Construct a AnalysisTest object.
17       */
18  
19      public AnalysisTest() {
20          super("JUnit test case for analysis of the AST");
21      }
22  
23      /**
24       * Run the compiler against each pass-test file under the folder specified
25       * by PASS_TESTS_DIR property.
26       */
27  
28      public void testPass() {
29          File passTestsDir = new File(System.getProperty("PASS_TESTS_DIR"));
30          String frontEnd = System.getProperty("FRONT_END");
31          File[] files = passTestsDir.listFiles();
32          boolean errorHasOccurred = false;
33          for (int i = 0; files != null && i < files.length; i++) {
34              if (files[i].toString().endsWith(".java")) {
35                  System.out.printf(
36                          "Running compiler (up to analysis) on %s ...\n\n",
37                          files[i].toString());
38                  String[] args = { "-a", files[i].toString() };
39                  Main.main(args);
40                  System.out.printf("\n\n");
41  
42                  // true even if a single test fails
43                  errorHasOccurred |= Main.errorHasOccurred();
44              }
45          }
46  
47          // We want all tests to pass
48          assertFalse(errorHasOccurred);
49      }
50  
51      /**
52       * Entry point.
53       * 
54       * @param args
55       *            command-line arguments.
56       */
57  
58      public static void main(String[] args) {
59          junit.textui.TestRunner.run(AnalysisTest.class);
60      }
61  
62  }
