5   // Prints to STDOUT all Fibonacci numbers below 10000.
6   
7   public class Fibonacci {
8   
9       // Entry point; prints to STDOUT all Fibonacci numbers below
10      // 10000.
11  
12      public static void main(String[] args) {
13          int lo = 0;
14          int hi = 1;
15          while (hi <= 9999) {
16              hi = hi + lo;
17              lo = hi - lo;
18              SPIM.printInt(lo);
19              SPIM.printChar('\n');
20          }
21      }
22  
23  }
