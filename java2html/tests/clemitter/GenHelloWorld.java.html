3   import jminusminus.CLEmitter;
4   import static jminusminus.CLConstants.*;
5   import java.util.ArrayList;
6   
7   /**
8    * This class programatically generates the class file for the 
9    * following Java application using CLEmitter:
10   * 
11   * public class HelloWorld
12   * {
13   *     private static String message()
14   *     {
15   *         return "Hello, World!"; 
16   *     }
17   *     
18   *     public static void main( String[] args ) 
19   *     {
20   *         System.out.println( HelloWorld.message() );
21   *     }
22   * }
23   */
24  
25  public class GenHelloWorld {
26  
27      public static void main(String[] args) {
28          CLEmitter e = new CLEmitter(true);
29          ArrayList<String> accessFlags = new ArrayList<String>();
30  
31          // Add HelloWorld class
32          accessFlags.add("public");
33          e.addClass(accessFlags, "HelloWorld", "java/lang/Object", null, true);
34  
35          // Add message() method to HelloWorld
36          accessFlags.clear();
37          accessFlags.add("private");
38          accessFlags.add("static");
39          e.addMethod(accessFlags, "message", "()Ljava/lang/String;", null, true);
40          e.addLDCInstruction("Hello, World!");
41          e.addNoArgInstruction(ARETURN);
42  
43          // Add main() method to HelloWorld
44          accessFlags.clear();
45          accessFlags.add("public");
46          accessFlags.add("static");
47          e.addMethod(accessFlags, "main", "([Ljava/lang/String;)V", null, true);
48          e.addMemberAccessInstruction(GETSTATIC, "java/lang/System", "out",
49                  "Ljava/io/PrintStream;");
50          e.addMemberAccessInstruction(INVOKESTATIC, "HelloWorld", "message",
51                  "()Ljava/lang/String;");
52          e.addMemberAccessInstruction(INVOKEVIRTUAL, "java/io/PrintStream",
53                  "println", "(Ljava/lang/String;)V");
54          e.addNoArgInstruction(RETURN);
55  
56          // Write HelloWorld.class to file system
57          e.write();
58      }
59  
60  }
