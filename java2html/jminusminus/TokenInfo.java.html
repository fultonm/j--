5   /**
6    * An enum of token kinds. Each entry in this enum represents the kind of a
7    * token along with its image (string representation).
8    * 
9    * When you add a new token to the scanner, you must also add an entry to this
10   * enum specifying the kind and image of the new token.
11   */
12  
13  enum TokenKind {
14      EOF("<EOF>"), ABSTRACT("abstract"), ASSERT("assert"), BOOLEAN("boolean"), BREAK("break"),
15      BYTE("byte"), CASE("case"), CATCH("catch"), CHAR("char"), CLASS("class"), CONST("const"),
16      CONTINUE("continue"), DEFAULT("default"), DOUBLE("double"), DO("do"), ELSE("else"),
17      ENUM("enum"), EXTENDS("extends"), FALSE("false"), FINAL("final"), FINALLY("finally"),
18      FLOAT("float"), FOR("for"), GOTO("goto"), IF("if"), IMPLEMENTS("implements"), IMPORT("import"),
19      INSTANCEOF("instanceof"), INT("int"), INTERFACE("interface"), LONG("long"), NATIVE("native"),
20      NEW("new"), NULL("null"), PACKAGE("package"), PRIVATE("private"), PROTECTED("protected"),
21      PUBLIC("public"), RETURN("return"), SHORT("short"), STATIC("static"), STRICTFP("strictfp"),
22      SUPER("super"), SWITCH("switch"), SYNCHRONIZED("synchronized"), THIS("this"), THROW("throw"),
23      THROWS("throws"), TRANSIENT("transient"), TRUE("true"), TRY("try"), VOID("void"),
24      VOLATILE("volatile"), WHILE("while"), PLUS("+"), ASSIGN("="), DEC("--"), EQUAL("=="), GT(">"),
25      INC("++"), LAND("&&"), LOR("||"), LE("<="), LT("<"), GE(">="), LNOT("!"), NOT_EQUAL("!="),
26      MINUS("-"), PLUS_ASSIGN("+="), MINUS_ASSIGN("-="), MULT_ASSIGN("*="), DIV_ASSIGN("/="),
27      MOD_ASSIGN("%="), BWLSHIFT_ASSIGN("<<="), BWRSHIFT_ASSIGN(">>="), BWAND_ASSIGN("&="),
28      BWXOR_ASSIGN("^="), BWOR_ASSIGN("|="), STAR("*"), DIV("/"), MOD("%"), LPAREN("("), RPAREN(")"),
29      LCURLY("{"), RCURLY("}"), LBRACK("["), RBRACK("]"), SEMI(";"), COMMA(","), DOT("."), BWAND("&"),
30      BWOR("|"), BWXOR("^"), BWCOMP("~"), BWLSHIFT("<<"), BWRSHIFT(">>"), BWZEROFILLRSHIFT(">>>"),
31      TERNARY_TEST("?"), TERNARY_ELSE(":"), IDENTIFIER("<IDENTIFIER>"), INT_LITERAL("<INT_LITERAL>"),
32      CHAR_LITERAL("<CHAR_LITERAL>"), STRING_LITERAL("<STRING_LITERAL>"),
33      DOUBLE_LITERAL("<DOUBLE_LITERAL>"), FLOAT_LITERAL("<FLOAT_LITERAL>"),
34      LONG_LITERAL("<LONG_LITERAL>");
35  
36      /** The token's string representation. */
37      private String image;
38  
39      /**
40       * Construct an instance TokenKind given its string representation.
41       * 
42       * @param image string representation of the token.
43       */
44  
45      private TokenKind(String image) {
46          this.image = image;
47      }
48  
49      /**
50       * Return the image of the token.
51       * 
52       * @return the token's image.
53       */
54  
55      public String image() {
56          return image;
57      }
58  
59      /**
60       * Return the string representation of the token.
61       * 
62       * @return the token's string representation.
63       */
64  
65      public String toString() {
66          return image;
67      }
68  
69  }

71  /**
72   * A representation of tokens returned by the lexical analyzer method,
73   * getNextToken(). A token has a kind identifying what kind of token it is, an
74   * image for providing any semantic text, and the line in which it occurred in
75   * the source file.
76   */
77  
78  class TokenInfo {
79  
80      /** Token kind. */
81      private TokenKind kind;
82  
83      /**
84       * Semantic text (if any). For example, the identifier name when the token kind
85       * is IDENTIFIER. For tokens without a semantic text, it is simply its string
86       * representation. For example, "+=" when the token kind is PLUS_ASSIGN.
87       */
88      private String image;
89  
90      /** Line in which the token occurs in the source file. */
91      private int line;
92  
93      /**
94       * Construct a TokenInfo from its kind, the semantic text forming the token, and
95       * its line number.
96       * 
97       * @param kind  the token's kind.
98       * @param image the semantic text comprising the token.
99       * @param line  the line in which the token occurs in the source file.
100      */
101 
102     public TokenInfo(TokenKind kind, String image, int line) {
103         this.kind = kind;
104         this.image = image;
105         this.line = line;
106     }
107 
108     /**
109      * Construct a TokenInfo from its kind, and its line number. Its image is simply
110      * its string representation.
111      * 
112      * @param kind the token's identifying number.
113      * @param line identifying the line on which the token was found.
114      */
115 
116     public TokenInfo(TokenKind kind, int line) {
117         this(kind, kind.toString(), line);
118     }
119 
120     /**
121      * Return the token's string representation.
122      * 
123      * @return the string representation.
124      */
125 
126     public String tokenRep() {
127         return kind.toString();
128     }
129 
130     /**
131      * Return the semantic text associated with the token.
132      * 
133      * @return the semantic text.
134      */
135 
136     public String image() {
137         return image;
138     }
139 
140     /**
141      * Return the line number associated with the token.
142      * 
143      * @return the line number.
144      */
145 
146     public int line() {
147         return line;
148     }
149 
150     /**
151      * Return the token's kind.
152      * 
153      * @return the kind.
154      */
155 
156     public TokenKind kind() {
157         return kind;
158     }
159 
160     /**
161      * Return the token's image.
162      * 
163      * @return the image.
164      */
165 
166     public String toString() {
167         return image;
168     }
169 
170 }
