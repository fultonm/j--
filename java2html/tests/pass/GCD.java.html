5   import java.lang.Integer;
6   import java.lang.System;
7   
8   // Accepts two ints (a and b) as command-line arguments and prints
9   // their greatest common divisor (GCD) to STDOUT.
10  
11  public class GCD {
12  
13      // Return the remainder of a and b from an implied division;
14      // a is the dividend and b is the divisor.
15  
16      private int mod(int a, int b) {
17          if (b == 0) {
18              return a;
19          } else if (a == b) {
20              return 0;
21          } else if (a > b) {
22              return mod(a - b, b);
23          }
24          return a;
25      }
26  
27      // Compute the GCD(a, b) using the Euclidean algorithm and
28      // return the value.
29  
30      public int compute(int a, int b) {
31          if (a == 0) {
32              return b;
33          }
34          return compute(b, mod(a, b));
35      }
36  
37      // Entry point; accepts two ints (a and b) as command-line
38      // arguments and prints GCD(a, b) to STDOUT.
39  
40      public static void main(String[] args) {
41          GCD gcd = new GCD();
42          int a = Integer.parseInt(args[0]);
43          int b = Integer.parseInt(args[1]);
44          System.out.println("GCD(" + a + ", " + b + ") = " + gcd.compute(a, b));
45      }
46  
47  }
